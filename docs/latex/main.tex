% --- CUSTOM MACROS ---

% 1. File or directory paths
% Usage: Open the \file{main.py} script.
\newcommand{\file}[1]{\textit{\texttt{\color{docDark}#1}}}

% 2. Inline code snippets or variables
% Usage: The \code{user_id} variable must be an integer.
\newcommand{\code}[1]{\colorbox{gray!10}{\texttt{\small\color{red!60!black}#1}}}

% 3. Class or Component names (PascalCase convention)
% Usage: The \class{CustomerFactory} instantiates the object.
\newcommand{\class}[1]{\textbf{\texttt{\color{docBlue}#1}}}

% 4. Keyboard shortcuts rendering
% Usage: Press \keys{Ctrl + C} to terminate.
\newcommand{\keys}[1]{%
  \tikz[baseline=(key.base)]
    \node[draw, fill=white, drop shadow={shadow xshift=0.2ex,shadow yshift=-0.2ex},
          rectangle, rounded corners=2pt, inner sep=1pt, line width=0.5pt,
          font=\scriptsize\sffamily](key) {#1};\kern1pt
}
